# SupportBot 分类规则

IF message_type == "WISMO":
  fetch_tracking(order_id)
  reply_template("tracking_update")
  
IF message_type == "return_request":
  IF days_since_delivery <= 30:
    approve_return()
    send_label()
  ELSE:
    escalate(marcus, priority="medium")

IF message_type == "product_question":
  search_faq(product_id)
  generate_response(faq_context)

DEFAULT:
  escalate(marcus, priority="high")
  include_full_context(True)

# InventoryPulse 补货触发

FOR EACH asin in tracked_asins:
  velocity = calc_avg_daily_sales(
    asin, window=30
  )
  days_left = current_stock / velocity
  
  # 旺季缓冲
  IF is_peak_season():
    threshold = 35  # 天
  ELSE:
    threshold = 21  # 天
  
  IF days_left < threshold:
    order_qty = calc_reorder_qty(
      velocity, lead_time, safety_stock
    )
    send_po_to_supplier(asin, order_qty)
    alert_marcus(asin, days_left, order_qty)
